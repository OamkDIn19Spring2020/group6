CREATE DATABASE group6;

USE group6;

GRANT ALL PRIVILEGES on group6.* to 'group6'@'localhost' IDENTIFIED BY 'group6_pass';


CREATE TABLE user_database(
    user_id INT primary key auto_increment,
    username VARCHAR(255),
    fname VARCHAR(255),
    lname VARCHAR(255),
    gender VARCHAR(255),
    birthyear SMALLINT,
    email VARCHAR(255),
    password VARCHAR(255) 
) Engine=InnoDB;


CREATE TABLE user_stats(
    stats_id INT primary key auto_increment,
    name VARCHAR (255),
    weight VARCHAR(3),
    height VARCHAR(3),
    age VARCHAR(3),
    gender VARCHAR(10),
    DateTime DATE
);

CREATE TABLE products_database(
    product_id INT primary key auto_increment,
    product_name VARCHAR(255),
    product_price VARCHAR(30),
    product_image VARCHAR(255)
) Engine=InnoDB;

CREATE TABLE purchase_database(
    purchase_id INT primary key auto_increment,
    product_id INT,
    username VARCHAR(255),
    product_name VARCHAR(255),
    product_price VARCHAR(30)
) Engine=InnoDB;
